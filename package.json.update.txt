# package.json scripts bölümüne eklenecek komutlar:

{
  "scripts": {
    "db:migrate": "prisma migrate dev",
    "db:migrate:deploy": "prisma migrate deploy", 
    "db:seed": "npx tsx prisma/seed.ts",
    "db:seed:simple": "npx tsx prisma/seed_simple_permissions.ts",
    "db:seed:clean": "npx tsx prisma/seed_simple_permissions.ts --clean",
    "db:reset": "prisma migrate reset --force",
    "db:studio": "prisma studio"
  }
}

# Kullanım:
# npm run db:seed:simple     -> Basit permission'ları ekle
# npm run db:migrate         -> Migration çalıştır  
# npm run db:studio          -> Database görsel editor
